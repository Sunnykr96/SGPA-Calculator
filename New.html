        function calculateSGPA() {

            const form = document.getElementById("sgpaForm");

            const credits = [4, 2, 3, 2, 4, 1, 2, 2, 2]; // Credit points for each subject

            let totalPoints = 0;

            let totalCredits = 22;

 

            const grades = Array.from(form.elements)

                .filter(el => el.tagName === "SELECT")

                .map(select => parseInt(select.value));

 

            if (grades.includes(NaN)) {

                document.getElementById("result").innerText = "Please fill all grades.";

                return;

            }

 

            grades.forEach((grade, index) => {

                totalPoints += grade * credits[index];

            });

 

            const sgpa = (totalPoints / totalCredits).toFixed(2);

            document.getElementById("result").innerText = `Your SGPA is: ${sgpa}`;

        }

    </script>
